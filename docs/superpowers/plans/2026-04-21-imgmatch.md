**Goal:** 构建一个 CLI 工具，在全屏截图中查找图案的坐标位置，支持多尺度模糊匹配。

**Architecture:** 单文件 Python CLI 工具，使用 OpenCV `matchTemplate` 进行多尺度模板匹配。对模板图按 0.5x~2.0x 生成缩放变体，在每个缩放级别执行 `TM_CCOEFF_NORMED` 匹配，取全局最高置信度的位置作为结果，输出匹配区域中心坐标。

**Tech Stack:** Python 3.11+, opencv-python, numpy

- [ ] **Step 1: 编写 find_template 函数**

```python
def find_template(source_path: str, template_path: str, scale_range=(0.5, 2.0), scale_step=0.05, threshold=0.6):
    """在截图中查找模板图案的最佳匹配位置。"""
    source = cv2.imread(source_path, cv2.IMREAD_GRAYSCALE)
    template = cv2.imread(template_path, cv2.IMREAD_GRAYSCALE)

    if source is None:
        print(f"错误: 无法读取截图文件: {source_path}", file=sys.stderr)
        sys.exit(1)
    if template is None:
        print(f"错误: 无法读取模板文件: {template_path}", file=sys.stderr)
        sys.exit(1)

    th, tw = template.shape[:2]
    sh, sw = source.shape[:2]

    if th > sh or tw > sw:
        print("错误: 模板图片比截图还大，无法匹配", file=sys.stderr)
        sys.exit(1)

    best_val = -1
    best_loc = None
    best_scale = 1.0

    scales = np.arange(scale_range[0], scale_range[1] + scale_step, scale_step)
    for scale in scales:
        scaled_h = int(th * scale)
        scaled_w = int(tw * scale)

        if scaled_h > sh or scaled_w > sw:
            continue
        if scaled_h < 3 or scaled_w < 3:
            continue

        scaled_template = cv2.resize(template, (scaled_w, scaled_h),
            interpolation=cv2.INTER_AREA if scale < 1 else cv2.INTER_CUBIC)

        result = cv2.matchTemplate(source, scaled_template, cv2.TM_CCOEFF_NORMED)
        _, max_val, _, max_loc = cv2.minMaxLoc(result)

        if max_val > best_val:
            best_val = max_val
            best_loc = max_loc
            best_scale = scale

    if best_val < threshold or best_loc is None:
        return None

    center_x = best_loc[0] + int(tw * best_scale / 2)
    center_y = best_loc[1] + int(th * best_scale / 2)

    return (center_x, center_y, best_val)
```

- [ ] **Step 2: 编写 CLI 入口 main 函数**

```python
def main():
    parser = argparse.ArgumentParser(description="在截图中查找图案的坐标位置")
    parser.add_argument("--source", "-s", required=True, help="全屏截图文件路径")
    parser.add_argument("--template", "-t", required=True, help="图案图片文件路径")
    parser.add_argument("--threshold", type=float, default=0.6, help="最低匹配置信度阈值 (默认: 0.6)")
    parser.add_argument("--scale-min", type=float, default=0.5, help="最小缩放比例 (默认: 0.5)")
    parser.add_argument("--scale-max", type=float, default=2.0, help="最大缩放比例 (默认: 2.0)")
    parser.add_argument("--scale-step", type=float, default=0.05, help="缩放步长 (默认: 0.05)")
    args = parser.parse_args()

    result = find_template(
        args.source, args.template,
        scale_range=(args.scale_min, args.scale_max),
        scale_step=args.scale_step,
        threshold=args.threshold,
    )

    if result is None:
        print("未找到匹配", file=sys.stderr)
        sys.exit(1)

    x, y, confidence = result
    print(f"x={x}, y={y}")
```
